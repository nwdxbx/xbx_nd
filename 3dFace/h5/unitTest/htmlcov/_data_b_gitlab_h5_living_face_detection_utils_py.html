def rand(a=0, b=1):

def get_random_data(annotation_line, input_shape, random=True, max_boxes=20, jitter=.3, hue=.1, sat=1.5, val=1.5, proc_img=True):

h, w = input_shape

box[:, [1,3]] = box[:, [1,3]]*scale + dy
